sequenceDiagram

1.3 代码实现

  1. 项目初始化

    # 创建项目
cargo new mini-rCore
cd mini-rCore

    # 配置Cargo.toml
[package]
name = "mini-rCore"
version = "0.1.0"
edition = "2021"

[dependencies]
riscv = { git = "https://github.com/rcore-os/riscv", rev = "340e5bf" }

  2. 裸机入口配置

#![no_std] // 不使用rust标准库
#![no_main]// 取消主函数，避免初始化主函数时启动标准库内函数

// 自定义主函数入口，由entry.asm跳转
pub fn rust_main() -> ! {    
}


    # 设置内核执行环境
# 操作系统内核入口点的汇编代码
# 主要负责初始化栈指针并跳转到Rust主函数

### 定义代码段（可执行代码部分）
    .section .text.entry
    .globl _start
_start:
    # 栈顶指针存入sp寄存器
    la sp, boot_stack_top
    call rust_main

### 定义未初始化的栈内存空间
    .section .bss.stack
    .globl boot_stack_lower_bound
boot_stack_lower_bound:
    .space 4096 * 16
